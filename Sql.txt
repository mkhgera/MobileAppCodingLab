CREATE TABLE Attractions(
	attrID SERIAL PRIMARY KEY,
	attrName VARCHAR NOT NULL,
	attrImg bytea NOT NULL,
	attrDescription text NOT NULL,
	categoryID INT,
	cityID INT,
	CONSTRAINT attrCategory FOREIGN KEY(categoryID) REFERENCES Categories(categoryID),
	CONSTRAINT attrCity FOREIGN KEY(cityID) REFERENCES Cities(cityID)
);
